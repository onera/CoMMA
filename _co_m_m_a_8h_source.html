129template<
130 typename CoMMAIndexType,
131 typename CoMMAWeightType,
132 typename CoMMAIntType>
133void agglomerate_one_level(
134 // Dual graph:
135 const std::vector<CoMMAIndexType> &adjMatrix_row_ptr,
136 const std::vector<CoMMAIndexType> &adjMatrix_col_ind,
137 const std::vector<CoMMAWeightType> &adjMatrix_areaValues,
138 const std::vector<CoMMAWeightType> &volumes,
139
140 // Additional info about the mesh
141 const std::vector<std::vector<CoMMAWeightType>> &centers,
142 const std::vector<CoMMAWeightType> &priority_weights,
143 const std::vector<CoMMAIndexType> &anisotropicCompliantCells,
144 const std::vector<CoMMAIntType> &n_bnd_faces,
145
146 // Anisotropy related info
147 bool build_anisotropic_lines,
148 bool is_anisotropic,
149 bool odd_line_length,
150 CoMMAWeightType threshold_anisotropy,
151
152 // Seed ordering
153 CoMMASeedsPoolT seed_ordering_type,
154
155 // Outputs
156 std::vector<CoMMAIndexType> &fc_to_cc, // Out
157 std::vector<CoMMAIndexType> &agglomerationLines_Idx, // In & out
158 std::vector<CoMMAIndexType> &agglomerationLines, // In & out
159
160 // Tuning of the algorithms
161 bool correction,
162 CoMMAIntType dimension,
163 CoMMAIntType goal_card,
164 CoMMAIntType min_card,
165 CoMMAIntType max_card,
166 CoMMAAspectRatioT aspect_ratio = CoMMAAspectRatioT::DIAMETER_OVER_RADIUS,
167 CoMMAIntType singular_card_thresh = 1,
168 std::optional<CoMMAIndexType> max_cells_in_line = std::nullopt,
169 CoMMAIntType fc_choice_iter = 1,
170 CoMMANeighbourhoodT neighbourhood_type = CoMMANeighbourhoodT::EXTENDED
171) {
172 // NOTATION
173 //======================================
174 // fc = Fine Cell
175 // cc = Coarse Cell
176
177 // USEFUL SHORTCUTS
178 //======================================
179 using SeedsPoolType =
180 Seeds_Pool<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>;
181 using DualGraphType =
182 Dual_Graph<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>;
183 using CCContainerType =
184 Coarse_Cell_Container<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>;
185 using IsotropicPtr = std::unique_ptr<
186 Agglomerator_Isotropic<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>>;
187
188 // SANITY CHECKS
189 //======================================
190 CHECK_INT_TYPE(CoMMAIndexType, "first template argument");
191 CHECK_INT_TYPE(CoMMAIntType, "third template argument");
192 if (!(dimension == 2 || dimension == 3))
193 throw std::invalid_argument("CoMMA - Error: dimension must be 2 or 3");
194 if (min_card <= 1 || goal_card <= 1 || max_card <= 1)
195 throw std::invalid_argument(
196 "CoMMA - Error: Cardinalities must be greater than 1"
197 );
198 if (!(min_card <= goal_card && goal_card <= max_card))
199 throw std::invalid_argument(
200 "CoMMA - Error: Cardinalities must be in order (min <= goal <= max)"
201 );
202 if (fc_choice_iter < 1)
203 throw std::invalid_argument(
204 "CoMMA - Error: the number of iteration for the choice of the fine "
205 "cells must be at least 1"
206 );
207 if (fc_choice_iter > iter_agglo_max_iter)
208 throw std::invalid_argument(
209 "CoMMA - Error: the number of iteration for the choice of the fine "
210 "cells must be at most "
211 + std::to_string(iter_agglo_max_iter)
212 );
213 if (adjMatrix_row_ptr.empty()
214 || adjMatrix_row_ptr.back()
215 != static_cast<CoMMAIndexType>(adjMatrix_col_ind.size())
216 || adjMatrix_row_ptr.back()
217 != static_cast<CoMMAIndexType>(adjMatrix_areaValues.size()))
218 throw std::invalid_argument(
219 "CoMMA - Error: bad CRS graph (sizes do not match)"
220 );
221 if (is_anisotropic) {
222 if (build_anisotropic_lines) {
223 if (anisotropicCompliantCells.empty()) {
224 std::cout << "CoMMA - Warning: building anisotropic line requested, no "
225 "compliant cells provided. Switching off anisotropy."
226 << std::endl;
227 }
228 if (max_cells_in_line.has_value() && max_cells_in_line.value() <= 0) {
229 std::cout << "CoMMA - Requested a negative or null maximum number of "
230 "cells in line. Dropping the limit."
231 << std::endl;
232 max_cells_in_line = std::nullopt;
233 }
234 } else {
235 // Anisotropic lines are provided
236 if (agglomerationLines_Idx.size() < 2 || agglomerationLines.empty()) {
237 std::cout
238 << "CoMMA - Warning: usage of input anisotropic line requested, "
239 "but arguments are not enough / invalid to define them. "
240 "Switching off anisotropy."
241 << std::endl;
242 is_anisotropic = false;
243 } else if (agglomerationLines_Idx.back()
244 != static_cast<CoMMAIndexType>(agglomerationLines.size())) {
245 throw std::invalid_argument(
246 "CoMMA - Error: bad anisotropic lines definition (sizes do not "
247 "match)"
248 );
249 }
250 }
251 }
252 auto sing_thresh = singular_card_thresh;
253 if (singular_card_thresh <= 0) {
254 throw std::invalid_argument(
255 "CoMMA - Error: Threshold cardinality for singular cells should be "
256 "greater than zero"
257 );
258 }
259 if (singular_card_thresh >= min_card) {
260 std::cout
261 << "CoMMA - Warning: Threshold cardinality is equal or larger than "
262 "minimum cardinality. Changing it to this latter value."
263 << std::endl;
264 sing_thresh = min_card - 1;
265 }
266
267 // SIZES CAST
268 //======================================
269 const auto nb_fc = static_cast<CoMMAIndexType>(adjMatrix_row_ptr.size() - 1);
270
271 // BOUNDARY FACES
272 //======================================
273 // Sometimes partitioners give a number of boundary faces higher than the
274 // physical one. We fix this
275 const CoMMAIntType expected_max_n_bnd = dimension;
276 std::vector<CoMMAIntType> fixed_n_bnd_faces(n_bnd_faces.size());
277 std::replace_copy_if(
278 n_bnd_faces.begin(),
279 n_bnd_faces.end(),
280 fixed_n_bnd_faces.begin(),
281 [expected_max_n_bnd](auto n) { return n > expected_max_n_bnd; },
282 expected_max_n_bnd
283 );
284
285 // SEED POOL
286 //======================================
287 // Object providing the order of agglomeration
288 std::shared_ptr<SeedsPoolType> seeds_pool = nullptr;
289 switch (seed_ordering_type) {
290 case CoMMASeedsPoolT::BOUNDARY_PRIORITY:
291 seeds_pool = std::make_shared<Seeds_Pool_Boundary_Priority<
292 CoMMAIndexType,
293 CoMMAWeightType,
294 CoMMAIntType>>(fixed_n_bnd_faces, priority_weights, false);
295 break;
296 case CoMMASeedsPoolT::NEIGHBOURHOOD_PRIORITY:
297 seeds_pool = std::make_shared<Seeds_Pool_Neighbourhood_Priority<
298 CoMMAIndexType,
299 CoMMAWeightType,
300 CoMMAIntType>>(fixed_n_bnd_faces, priority_weights, false);
301 break;
302 case CoMMASeedsPoolT::BOUNDARY_PRIORITY_ONE_POINT_INIT:
303 seeds_pool = std::make_shared<Seeds_Pool_Boundary_Priority<
304 CoMMAIndexType,
305 CoMMAWeightType,
306 CoMMAIntType>>(fixed_n_bnd_faces, priority_weights, true);
307 break;
308 case CoMMASeedsPoolT::NEIGHBOURHOOD_PRIORITY_ONE_POINT_INIT:
309 seeds_pool = std::make_shared<Seeds_Pool_Neighbourhood_Priority<
310 CoMMAIndexType,
311 CoMMAWeightType,
312 CoMMAIntType>>(fixed_n_bnd_faces, priority_weights, true);
313 break;
314 default:
315 throw std::invalid_argument("CoMMA - Error: Seeds pool type unsupported");
316 }
317
318 // DUAL GRAPH
319 //======================================
320 // Object containing the graph representation and related info in a convenient
321 // structure
322 std::shared_ptr<DualGraphType> fc_graph = std::make_shared<DualGraphType>(
323 nb_fc,
324 adjMatrix_row_ptr,
325 adjMatrix_col_ind,
326 adjMatrix_areaValues,
327 volumes,
328 centers,
329 fixed_n_bnd_faces,
330 dimension,
331 anisotropicCompliantCells
332 );
333
334 // COARSE CELL CONTAINER
335 //======================================
336 // Preparing the object that will contain all the coarse cells
337 std::shared_ptr<CCContainerType> cc_graph =
338 std::make_shared<CCContainerType>(fc_graph, sing_thresh);
339
340 // AGGLOMERATION OF ANISOTROPIC CELLS
341 //======================================
342 // @todo maybe re-refactor the class agglomerator to allow the implicit upcast
343 // like the biconnected case
344 // The agglomerator anisotropic is not called with the implicit upcasting
345 // pointing because of the initialization of
346 // the anisotropic lines.
347 // for more information look at:
348 // https://stackoverflow.com/questions/19682402/initialize-child-object-on-a-pointer-to-parent
349 // About constructors when upcasting:
350 // https://www.reddit.com/r/learnprogramming/comments/1wopf6/java_which_constructor_is_called_when_upcasting/
351 if (is_anisotropic) {
352 // Build anisotropic agglomerator
353 Agglomerator_Anisotropic<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>
354 aniso_agg(
355 fc_graph,
356 cc_graph,
357 seeds_pool,
358 dimension,
359 threshold_anisotropy,
360 agglomerationLines_Idx,
361 agglomerationLines,
362 priority_weights,
363 build_anisotropic_lines,
364 odd_line_length,
365 max_cells_in_line
366 );
367
368 // Agglomerate anisotropic cells only
369 aniso_agg.agglomerate_one_level(
370 goal_card, min_card, max_card, priority_weights, false
371 );
372
373 // Put anisotropic lines into the output parameters
374 // (Info about level of the line: WARNING! here 1 it means that we give it
375 // back lines in the new global index, 0 the old)
376 aniso_agg.export_anisotropic_lines(
377 1, agglomerationLines_Idx, agglomerationLines
378 );
379 } else {
380 seeds_pool->initialize();
381 }
382
383 // AGGLOMERATION OF ISOTROPIC CELLS
384 //======================================
385 // We define here the type of Agglomerator
386 IsotropicPtr agg = nullptr;
387 // TODO: maybe pass to a switch when another agglomerator will be implemented
388 if (fc_choice_iter > 1) {
389 agg = std::make_unique<
390 Agglomerator_Iterative<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>>(
391 fc_graph,
392 cc_graph,
393 seeds_pool,
394 dimension,
395 aspect_ratio,
396 neighbourhood_type,
397 fc_choice_iter
398 );
399 } else {
400 agg = std::make_unique<
401 Agglomerator_Biconnected<CoMMAIndexType, CoMMAWeightType, CoMMAIntType>>(
402 fc_graph,
403 cc_graph,
404 seeds_pool,
405 dimension,
406 aspect_ratio,
407 neighbourhood_type,
408 fc_choice_iter
409 );
410 }
411 // Agglomerate
412 agg->agglomerate_one_level(
413 goal_card, min_card, max_card, priority_weights, correction
414 );
415 // FILLING FC TO CC (it is a property of the cc_graph but retrieved through an
416 // helper of the agglomerator)
417 fc_to_cc.clear();
418 fc_to_cc.reserve(cc_graph->_fc_2_cc.size());
419 std::transform(
420 cc_graph->_fc_2_cc.begin(),
421 cc_graph->_fc_2_cc.end(),
422 std::back_inserter(fc_to_cc),
423 [](const auto &f2c) { return f2c.value(); }
424 );
425}

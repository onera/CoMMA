81template<typename T>
82inline T get_direction(
83 const std::vector<T> &a, const std::vector<T> &b, std::vector<T> &dir
84) {
85 T norm{0.};
86 for (auto i = decltype(a.size()){0}; i < a.size(); ++i) {
87 const T di = b[i] - a[i];
88 dir[i] = di;
89 norm += di * di;
90 }
91 norm = sqrt(norm);
92 const T ov_norm = T{1.} / norm;
93 for (auto &di : dir)
94 di *= ov_norm;
95 return norm;
96}

308template<typename IndexT, typename DistT>
309void compute_neighbourhood_based_wall_distance(
310 const std::vector<IndexT> &neigh_idxs,
311 const std::vector<IndexT> &neighs,
312 const std::vector<IndexT> &wall,
313 std::vector<DistT> &dist
314) {
315 static_assert(
316 std::is_signed<DistT>::value,
317 "The distance type should be signed to allow flags (negative values)"
318 );
319 dist.resize(neigh_idxs.size() - 1);
320 std::fill(dist.begin(), dist.end(), DistT{-1});
321 std::queue<IndexT> to_visit{};
322 for (const auto &cell : wall) {
323 dist[cell] = DistT{0};
324 to_visit.emplace(cell);
325 }
326 while (!to_visit.empty()) {
327 // If you get wrong results, this could might be to the fact that one should
328 // consider not specially the first of the queue, but the one with lowest
329 // distance visited so far. So, the definition of `cur` below should be
330 // changed, and with that, another container should be used instead of a
331 // queue, since one might pop from whenever and not only the front (a deque
332 // should work).
333 const auto cur = to_visit.front();
334 to_visit.pop();
335 const auto cur_dist = dist[cur] + DistT{1};
336 for (auto neigh = neighs.cbegin() + neigh_idxs[cur];
337 neigh < neighs.cbegin() + neigh_idxs[cur + 1];
338 ++neigh) {
339 if (dist[*neigh] < DistT{0}) {
340 dist[*neigh] = cur_dist;
341 to_visit.emplace(*neigh);
342 } else if (dist[*neigh] > cur_dist) {
343 dist[*neigh] = cur_dist;
344 }
345 }
346 }
347}

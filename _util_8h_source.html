67template<unsigned int p, typename T>
68inline T constexpr int_power(const T x) {
69 // See https://stackoverflow.com/a/1505791/12152457
70 if constexpr (p == 0) return 1;
71 if constexpr (p == 1) return x;
72 if constexpr (p == 2) return x * x;
73 if constexpr (p == 3) return x * x * x;
74
75 constexpr int tmp = int_power<p / 2>(x);
76 if constexpr ((p % 2) == 0) {
77 return tmp * tmp;
78 } else {
79 return x * tmp * tmp;
80 }
81}

123template<typename T>
124inline T get_direction(
125 const std::vector<T> &a, const std::vector<T> &b, std::vector<T> &dir
126) {
127 T norm{0.};
128 for (auto i = decltype(a.size()){0}; i < a.size(); ++i) {
129 const T di = b[i] - a[i];
130 dir[i] = di;
131 norm += di * di;
132 }
133 norm = sqrt(norm);
134 const T ov_norm = T{1.} / norm;
135 for (auto &di : dir)
136 di *= ov_norm;
137 return norm;
138}

306template<typename IndexT, typename IntT>
307inline void filter_cells_by_n_edges(
308 const std::vector<IndexT> &indices,
309 const std::vector<IntT> &n_bnd_faces,
310 const std::unordered_set<IntT> &allowed,
311 std::vector<IndexT> &filtered
312) {
313 filtered.clear();
314 auto target = decltype(allowed){};
315 std::copy_if(
316 allowed.begin(),
317 allowed.end(),
318 std::inserter(target, target.begin()),
319 [](const auto n) { return n > 0; }
320 );
321 if (!target.empty()) {
322 const auto n_cells = static_cast<IndexT>(n_bnd_faces.size());
323 filtered.reserve(n_cells);
324 for (auto c = decltype(n_cells){0}; c < n_cells; ++c) {
325 if (allowed.find(
326 static_cast<IntT>(indices[c + 1] - indices[c] + n_bnd_faces[c])
327 )
328 != allowed.end())
329 filtered.emplace_back(c);
330 }
331 filtered.shrink_to_fit();
332 }
333}

356template<typename IndexT, typename DistT>
357void compute_neighbourhood_based_wall_distance(
358 const std::vector<IndexT> &neigh_idxs,
359 const std::vector<IndexT> &neighs,
360 const std::vector<IndexT> &wall,
361 std::vector<DistT> &dist
362) {
363 static_assert(
364 std::is_signed<DistT>::value,
365 "The distance type should be signed to allow flags (negative values)"
366 );
367 dist.resize(neigh_idxs.size() - 1);
368 std::fill(dist.begin(), dist.end(), DistT{-1});
369 std::queue<IndexT> to_visit{};
370 for (const auto &cell : wall) {
371 dist[cell] = DistT{0};
372 to_visit.emplace(cell);
373 }
374 while (!to_visit.empty()) {
375 // If you get wrong results, this could might be to the fact that one should
376 // consider not specially the first of the queue, but the one with lowest
377 // distance visited so far. So, the definition of `cur` below should be
378 // changed, and with that, another container should be used instead of a
379 // queue, since one might pop from whenever and not only the front (a deque
380 // should work).
381 const auto cur = to_visit.front();
382 to_visit.pop();
383 const auto cur_dist = dist[cur] + DistT{1};
384 for (auto neigh = neighs.cbegin() + neigh_idxs[cur];
385 neigh < neighs.cbegin() + neigh_idxs[cur + 1];
386 ++neigh) {
387 if (dist[*neigh] < DistT{0}) {
388 dist[*neigh] = cur_dist;
389 to_visit.emplace(*neigh);
390 } else if (dist[*neigh] > cur_dist) {
391 dist[*neigh] = cur_dist;
392 }
393 }
394 }
395}

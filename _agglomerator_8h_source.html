49template<typename IndexT, typename RealT, typename IntT, int dim>
50inline RealT compute_AR_max_ov_radius(
51 const CellFeatures<IndexT, RealT, IntT> &feat
52) {
53 if constexpr (dim < 2) {
54 return sqrt(feat._sq_diam) / feat._measure;
55 } else if constexpr (dim == 2) {
56 return sqrt(feat._sq_diam) / sqrt(feat._measure);
57 } else if constexpr (dim == 3) {
58 return sqrt(feat._sq_diam) / cbrt(feat._measure);
59 } else {
60 return sqrt(feat._sq_diam) / pow(feat._measure, 1.0 / dim);
61 }
62}

335 void agglomerate_one_level(
336 const CoMMAIntType goal_card,
337 const CoMMAIntType min_card,
338 const CoMMAIntType max_card,
339 const std::vector<CoMMAWeightType> &priority_weights,
340 bool correction_steps
341 ) override {
342 // Unused parameters
343 CoMMAUnused(goal_card);
344 CoMMAUnused(min_card);
345 CoMMAUnused(max_card);
346 CoMMAUnused(correction_steps);
347
348 if (this->_should_agglomerate) {
349 // Necessary for the create_cc but maybe we need in
350 // some way to change that.
351 constexpr bool is_anisotropic = true;
352 constexpr CoMMAIntType compactness = 1;
353
354 // Setting up lambda function that will perform the loop. This is needed
355 // since we want to loop forwards or backwards according to some runtime
356 // value See, e.g., https://stackoverflow.com/a/56133699/12152457
357 auto loop_line = [&](auto begin, auto end) {
358 AnisotropicLinePtr line_lvl_p_one = std::make_shared<AnisotropicLine>();
359 typename decltype(this->_max_cells_in_line)::value_type n_cells{0};
360 for (auto line_it = begin; line_it != end; line_it += 2) {
361 if (this->_max_cells_in_line.has_value()
362 && n_cells + 2 > this->_max_cells_in_line.value()) {
363 // Max length reached, break
364 break;
365 }
366 // we agglomerate cells along the agglomeration line, hence we have to
367 // go through the faces and agglomerate two faces together
368 // Here we have to consider a special case when we have an odd number
369 // of cells: THIS IS FUNDAMENTAL FOR THE CONVERGENCE OF THE MULTIGRID
370 // ALGORITHM
371 std::unordered_set<CoMMAIndexType> s_fc = {*line_it, *(line_it + 1)};
372 n_cells += 2;
373 // We update the neighbours. At this stage, we do not check if it is
374 // or will be agglomerated since there will be a cleaning step after
375 // the anisotropic agglomeration
376 this->_aniso_neighbours.insert(
377 this->_aniso_neighbours.end(),
378 this->_fc_graph->neighbours_cbegin(*line_it),
379 this->_fc_graph->neighbours_cend(*line_it)
380 );
381 this->_aniso_neighbours.insert(
382 this->_aniso_neighbours.end(),
383 this->_fc_graph->neighbours_cbegin(*(line_it + 1)),
384 this->_fc_graph->neighbours_cend(*(line_it + 1))
385 );
386 if (
387 std::distance(line_it, end) == 3
388 || (
389 this->_max_cells_in_line.has_value()
390 && n_cells + 1 == this->_max_cells_in_line.value())) {
391 if (this->_odd_line_length) {
392 // If only three cells left, agglomerate them
393 s_fc.insert(*(line_it + 2));
394 n_cells++;
395 this->_aniso_neighbours.insert(
396 this->_aniso_neighbours.end(),
397 this->_fc_graph->neighbours_cbegin(*(line_it + 2)),
398 this->_fc_graph->neighbours_cend(*(line_it + 2))
399 );
400 }
401 line_it++; // Ensure to break the loop after current iteration
402 }
403 // We create the coarse cell
404 const CoMMAIndexType i_cc =
405 this->_cc_graph->create_cc(s_fc, compactness, is_anisotropic);
406 line_lvl_p_one->push_back(i_cc);
407 }
408
409 this->_v_lines[1].push_back(line_lvl_p_one);
410 }; // End lambda def
411
412 // Process of every agglomeration lines:
413 for (auto line_ptr = this->_v_lines[0].begin();
414 line_ptr != this->_v_lines[0].end();
415 line_ptr++) {
416 // We iterate on the anisotropic lines of a particular level (the level
417 // 1, where they were copied from level 0). We create a pointer to an
418 // empty deque for the line + 1, and hence for the next level of
419 // agglomeration
420 auto line = *line_ptr;
421 if (line->size() <= 1) {
422 // the agglomeration_line is empty and hence the iterator points again
423 // to the empty deque, updating what is pointed by it and hence
424 // __v_lines[1] (each time we iterate on the line, a new deque
425 // line_lvl_p_one is defined)
426 continue;
427 }
428 // We start agglomerating from the head or the tail of the line
429 // according to which of the two has more boundary faces, or priority or
430 // ID
431 const auto l_fr = line->front(), l_bk = line->back();
432 const auto bnd_fr = this->_fc_graph->get_n_boundary_faces(l_fr),
433 bnd_bk = this->_fc_graph->get_n_boundary_faces(l_bk);
434 const auto w_fr = priority_weights[l_fr], w_bk = priority_weights[l_bk];
435 const bool forward_line =
436 bnd_fr > bnd_bk // Greater boundaries...
437 || (bnd_fr == bnd_bk &&
438 (w_fr > w_bk // ...or greater priority...
439 || (w_fr == w_bk && l_fr < l_bk) // ..or smaller index
440 ));
441
442 if (forward_line)
443 loop_line(line->cbegin(), line->cend());
444 else
445 loop_line(line->crbegin(), line->crend());
446
447 } // End loop on lines
448
449 this->update_seeds_pool();
450
451 } // End if should agglomerate
452 else
453 // If it didn't agglomerate, initialize for sure
454 this->_seeds_pool->initialize();
455 }

460 void update_seeds_pool() {
461 if (!this->_aniso_neighbours.empty()) {
462 // Example of erase taken from
463 // https://en.cppreference.com/w/cpp/container/deque/erase
464 for (auto it = this->_aniso_neighbours.begin();
465 it != this->_aniso_neighbours.end();) {
466 if (this->_cc_graph->_is_fc_agglomerated[*it])
467 it = this->_aniso_neighbours.erase(it);
468 else
469 ++it;
470 }
471 if (!this->_aniso_neighbours.empty()) {
472 // The master queue is the one of the first agglomerated cell:
473 // It is important to set it in the case the user asked for
474 // neighbourhood priority
475 this->_seeds_pool->set_top_queue(
476 this->_fc_graph->get_n_boundary_faces(this->_aniso_neighbours.front())
477 );
478 this->_seeds_pool->update(this->_aniso_neighbours);
479 }
480 }
481 // Even if we have updated it, the seeds pool might need initialization, for
482 // instance, if it was set up with boundary priority
483 if (this->_seeds_pool->need_initialization(
484 this->_cc_graph->_is_fc_agglomerated
485 ))
486 this->_seeds_pool->initialize();
487 }

496 void export_anisotropic_lines(
497 CoMMAIntType level,
498 std::vector<CoMMAIndexType> &aniso_lines_idx,
499 std::vector<CoMMAIndexType> &aniso_lines
500 ) const {
501 // Reset lines
502 aniso_lines_idx.clear();
503 aniso_lines.clear();
504 if (this->_should_agglomerate && !this->_v_lines[level].empty()) {
505 // If at the level of agglomeration "level" the vector containing the
506 // number of lines is empty, hence it means no line has been found at the
507 // current level. variable cumulating the number of fine cells in the
508 // agglomeration lines of the current level
509 CoMMAIndexType number_of_fc_in_agglomeration_lines = 0;
510 aniso_lines_idx.push_back(0);
511 // We cycle over the line (in _v_lines)
512 for (const auto &line_ptr : this->_v_lines[level]) {
513 const CoMMAIndexType size_of_line = line_ptr->size();
514 // This vector store the end of a line and the start of a new
515 // anisotropic line WARNING! We are storing the anisotropic lines in a
516 // vector so we need a way to point to the end of a line and the
517 // starting of a new one.
518 aniso_lines_idx.push_back(
519 size_of_line + number_of_fc_in_agglomeration_lines
520 );
521 // Here we store the index of the cell.
522 for (const auto cell : *line_ptr) {
523 aniso_lines.push_back(cell);
524 }
525 number_of_fc_in_agglomeration_lines += size_of_line;
526 }
527 }

546protected:
558 bool build_anisotropic_lines(
559 const std::vector<CoMMAWeightType> &priority_weights,
560 const CoMMAWeightType threshold_anisotropy
561 ) {
562 std::deque<CoMMAIndexType> aniso_seeds_pool;
563 // It is the max_weight, hence the maximum area among the faces composing
564 // the cell. Used to recognized the face
565 std::vector<CoMMAWeightType> max_weights(
566 this->_fc_graph->_number_of_cells, 0.0
567 );
568 std::vector<bool> to_treat(this->_fc_graph->_number_of_cells, false);
569 // Computation of the anisotropic cell, alias of the cells for which the
570 // ratio between the face with maximum area and the face with minimum area
571 // is more than a given threshold.
572 this->_fc_graph->tag_anisotropic_cells(
573 max_weights,
574 to_treat,
575 aniso_seeds_pool,
576 threshold_anisotropy,
577 priority_weights,
578 0
579 );
580 if (aniso_seeds_pool.empty()) {
581 std::cout << "CoMMA - No anisotropic cell found. Skipping anisotropic "
582 "agglomeration"
583 << std::endl;
584 return false;
585 }
586 // Size might not be the dimension
587 const auto pts_dim = this->_fc_graph->_centers[0].size();
588 // size of the line
589 this->_nb_lines[0] = 0;
590 // we cycle on all the anisotropic cells identified before
591 for (auto &i_fc : aniso_seeds_pool) {
592 // seed from where we start building the line
593 if (!to_treat[i_fc]) {
594 // If the cell has been already treated, continue to the next
595 // anisotropic cell
596 continue;
597 }
598 // we save the primal seed for the opposite direction check that will
599 // happen later
600 const auto primal_seed = i_fc;
601 std::optional<std::vector<CoMMAWeightType>> primal_dir = std::nullopt;
602 // seed to be considered to add or not a new cell to the line
603 CoMMAIndexType seed = primal_seed;
604 // Create the new line
605 AnisotropicLinePtr cur_line = std::make_shared<AnisotropicLine>();
606 // we add the first seed
607 cur_line->push_back(seed);
608 to_treat[seed] = false;
609 // Flag to determine end of line
610 bool end = false;
611 // Flag to determine if we arrived at the end of an extreme of a line
612 bool opposite_direction_check = false;
613 // Info about growth direction
614 std::vector<CoMMAWeightType> prev_dir(pts_dim);
615 bool empty_line = true;
616 // Start the check from the seed
617 // while the line is not ended
618 while (!end) {
619 // for the seed (that is updated each time end!= true) we fill the
620 // neighbours and the weights
621 auto n_it = this->_fc_graph->neighbours_cbegin(seed);
622 auto w_it = this->_fc_graph->weights_cbegin(seed);
623 // std::vector of the candidates to continue the line
624 CoMMASetOfPairType candidates;
625 // If the line is long enough, we use the direction. Otherwise, we use
626 // the weight. Putting a high-level if to reduce the branching inside
627 // the loop over the neighbours.
628 if (empty_line) {
629 for (; n_it != this->_fc_graph->neighbours_cend(seed);
630 ++n_it, ++w_it) {
631 if (to_treat[*n_it] && *w_it > 0.90 * max_weights[seed]) {
632 candidates.emplace(*n_it, *w_it);
633 }
634 } // end for loop
635 } else {
636 // If not an empty line, we check the direction, see
637 // !dot_deviate below
638 for (; n_it != this->_fc_graph->neighbours_cend(seed);
639 ++n_it, ++w_it) {
640 if (to_treat[*n_it] && *w_it > 0.90 * max_weights[seed]) {
641 std::vector<CoMMAWeightType> cur_dir(pts_dim);
642 get_direction<CoMMAWeightType>(
643 this->_fc_graph->_centers[seed],
644 this->_fc_graph->_centers[*n_it],
645 cur_dir
646 );
647 const auto dot = dot_product<CoMMAWeightType>(prev_dir, cur_dir);
648 if (!dot_deviate<CoMMAWeightType>(dot)) {
649 candidates.emplace(*n_it, fabs(dot));
650 }
651 }
652 } // end for loop
653 }
654 if (!candidates.empty()) {
655 // Even if we have more than one candidate, we choose just one
656 // otherwise we risk to add 2 (problematic with primal seed)
657 // It is what is done in Mavriplis
658 // https://scicomp.stackexchange.com/questions/41830/anisotropic-lines-identification-algorithm
662 // update the seed to the actual candidate
663 const auto old_seed = seed;
664 seed = candidates.begin()->first;
665 if (!opposite_direction_check) {
666 cur_line->push_back(seed);
667 } else {
668 cur_line->push_front(seed);
669 }
670 to_treat[seed] = false;
671 empty_line = false;
672 get_direction<CoMMAWeightType>(
673 this->_fc_graph->_centers[old_seed],
674 this->_fc_graph->_centers[seed],
675 prev_dir
676 );
677 if (!primal_dir.has_value()) primal_dir = prev_dir;
678 }
679 // 0 candidate, we are at the end of the line or at the end of one
680 // direction
681 else /*if (candidates.size() == 0)*/ {
682 // Before giving up, let's try another thing: Doing the same things as
683 // above with the only difference that we allow to move through any
684 // faces and not only the maximum one but still checking for direction
685 if (!empty_line) {
686 // If not an empty line, we check the direction, see is_parallel
687 // below
688 for (auto it = this->_fc_graph->neighbours_cbegin(seed);
689 it != this->_fc_graph->neighbours_cend(seed);
690 ++it) {
691 if (to_treat[*it]) {
692 std::vector<CoMMAWeightType> cur_dir(pts_dim);
693 get_direction<CoMMAWeightType>(
694 this->_fc_graph->_centers[seed],
695 this->_fc_graph->_centers[*it],
696 cur_dir
697 );
698 const auto dot =
699 dot_product<CoMMAWeightType>(prev_dir, cur_dir);
700 if (!dot_deviate<CoMMAWeightType>(dot)) {
701 candidates.emplace(*it, fabs(dot));
702 }
703 }
704 } // end for loop
705 if (!candidates.empty()) {
706 // We found one! Keep going!
707 const auto old_seed = seed;
708 seed = candidates.begin()->first;
709 if (!opposite_direction_check) {
710 cur_line->push_back(seed);
711 } else {
712 cur_line->push_front(seed);
713 }
714 to_treat[seed] = false;
715 empty_line = false;
716 get_direction<CoMMAWeightType>(
717 this->_fc_graph->_centers[old_seed],
718 this->_fc_graph->_centers[seed],
719 prev_dir
720 );
721 if (!primal_dir.has_value()) primal_dir = prev_dir;
722 } else {
723 // If we have already looked at the other side of the line or if
724 // the primal seed is where we are already at, there is nothing
725 // that we can do
726 if (!opposite_direction_check && seed != primal_seed) {
727 seed = primal_seed;
728 // The check seed != primal_seed should ensure that
729 // primal_dir != null
730 prev_dir = primal_dir.value();
731 opposite_direction_check = true;
732 } else {
733 end = true;
734 }
735 }
736 } // End last step check on neighbours
737 else {
738 // If we have already looked at the other side of the line or if
739 // the primal seed is where we are already at, there is nothing
740 // that we can do
741 if (!opposite_direction_check && seed != primal_seed) {
742 seed = primal_seed;
743 // The check seed != primal_seed should ensure that
744 // primal_dir != null
745 prev_dir = primal_dir.value();
746 opposite_direction_check = true;
747 } else {
748 end = true;
749 }
750 }
751 } // End of no candidates case
752 } // End of a line
753 // we push the deque to the list if are bigger than 1
754 if (cur_line->size() > 1) {
755 this->_v_lines[0].push_back(cur_line);
756 this->_nb_lines[0] += 1;
757 }
758 } // End of loop over anisotropic cells
759
760 if (this->_nb_lines[0] == 0) {
761 std::cout << "CoMMA - No anisotropic line was built (e.g., only isolated "
762 "anisotropic cells). Skipping anisotropic agglomeration"
763 << std::endl;
764 return false;
765 }
766 return true;
767 }
